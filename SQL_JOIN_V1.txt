1)a)	SELECT CUSTOMERS.FIRST_NAME,CUSTOMERS.LAST_NAME,PACKAGES.PACK_ID,PACKAGES.SPEED 
	FROM CUSTOMERS INNER JOIN PACKAGES 
	ON CUSTOMERS.PACK_ID=PACKAGES.PACK_ID 
  b)    SELECT CUSTOMERS.FIRST_NAME,CUSTOMERS.LAST_NAME,PACKAGES.PACK_ID,PACKAGES.SPEED 
	FROM CUSTOMERS INNER JOIN PACKAGES 
	ON CUSTOMERS.PACK_ID=PACKAGES.PACK_ID AND ( CUSTOMERS.PACK_ID=22 OR CUSTOMERS.PACK_ID=27)ORDER BY LAST_NAME
2)a)    SELECT PACKAGES.PACK_ID,PACKAGES.SPEED,PACKAGES.MONTHLY_PAYMENT,SECTORS.SECTOR_NAME 
	FROM PACKAGES INNER JOIN SECTORS 
	ON PACKAGES.SECTOR_ID=SECTORS.SECTOR_ID
  b)    SELECT CUSTOMERS.FIRST_NAME||' '||CUSTOMERS.LAST_NAME AS CUSTOMER_NAME,PACKAGES.PACK_ID,PACKAGES.SPEED,PACKAGES.MONTHLY_PAYMENT,SECTORS.SECTOR_NAME
	FROM CUSTOMERS INNER JOIN PACKAGES
	ON CUSTOMERS.PACK_ID=PACKAGES.PACK_ID 
	INNER JOIN SECTORS 
	ON PACKAGES.SECTOR_ID=SECTORS.SECTOR_ID
  c)    SELECT CUSTOMERS.FIRST_NAME||' '||CUSTOMERS.LAST_NAME AS CUSTOMER_NAME,PACKAGES.PACK_ID,PACKAGES.SPEED,PACKAGES.MONTHLY_PAYMENT,SECTORS.SECTOR_NAME
	FROM CUSTOMERS INNER JOIN PACKAGES
	ON CUSTOMERS.PACK_ID=PACKAGES.PACK_ID 
	INNER JOIN SECTORS 
	ON PACKAGES.SECTOR_ID=SECTORS.SECTOR_ID AND SECTORS.SECTOR_NAME='Business'
3)	SELECT CUSTOMERS.LAST_NAME,CUSTOMERS.FIRST_NAME,CUSTOMERS.JOIN_DATE,CUSTOMERS.PACK_ID,PACKAGES.SPEED,SECTORS.SECTOR_NAME
	FROM CUSTOMERS INNER JOIN PACKAGES
	ON CUSTOMERS.PACK_ID=PACKAGES.PACK_ID AND EXTRACT(YEAR FROM JOIN_DATE)='2006'
	INNER JOIN SECTORS
	ON PACKAGES.SECTOR_ID=SECTORS.SECTOR_ID AND SECTORS.SECTOR_NAME='Private'

4)	SELECT PACKAGES.PACK_ID,PACKAGES.SPEED,PACKAGES.MONTHLY_PAYMENT,PACK_GRADES.GRADE_NAME
	FROM PACKAGES INNER JOIN PACK_GRADES ON PACKAGES.MONTHLY_PAYMENT BETWEEN PACK_GRADES.MIN_PRICE AND PACK_GRADES.MAX_PRICE 
5)a)	SELECT CUSTOMERS.FIRST_NAME,CUSTOMERS.LAST_NAME,PACKAGES.SPEED,PACKAGES.MONTHLY_PAYMENT
	FROM CUSTOMERS INNER JOIN PACKAGES
	ON CUSTOMERS.PACK_ID=PACKAGES.PACK_ID
  b)    SELECT CUSTOMERS.FIRST_NAME,CUSTOMERS.LAST_NAME,PACKAGES.SPEED,PACKAGES.MONTHLY_PAYMENT
	FROM CUSTOMERS LEFT OUTER JOIN PACKAGES
	ON CUSTOMERS.PACK_ID=PACKAGES.PACK_ID
  c)    SELECT CUSTOMERS.FIRST_NAME,CUSTOMERS.LAST_NAME,PACKAGES.SPEED,PACKAGES.MONTHLY_PAYMENT
	FROM CUSTOMERS RIGHT OUTER JOIN PACKAGES
	ON CUSTOMERS.PACK_ID=PACKAGES.PACK_ID
  d)    SELECT CUSTOMERS.FIRST_NAME,CUSTOMERS.LAST_NAME,PACKAGES.SPEED,PACKAGES.MONTHLY_PAYMENT
	FROM CUSTOMERS FULL OUTER JOIN PACKAGES
	ON CUSTOMERS.PACK_ID=PACKAGES.PACK_ID
6)	SELECT C1.LAST_NAME,C1.FIRST_NAME,C2.PACK_ID
	FROM CUSTOMERS C1 JOIN CUSTOMERS C2
	ON C1.PACK_ID=C2.PACK_ID AND C2.FIRST_NAME='Amado' AND C2.LAST_NAME='Taylor'
7)	SELECT C1.LAST_NAME,C1.FIRST_NAME,C1.MONTHLY_DISCOUNT
	FROM CUSTOMERS C1 JOIN CUSTOMERS C2
	ON C1.MONTHLY_DISCOUNT<C2.MONTHLY_DISCOUNT AND C2.CUSTOMER_ID=103

8)	SELECT P1.PACK_ID,P1.SPEED
	FROM PACKAGES P1 JOIN PACKAGES P2
	ON P1.SPEED=P2.SPEED AND P2.PACK_ID=10
