----2------
SELECT COUNTRY_ID,COUNTRY_NAME FROM COUNTRIES
MINUS
(SELECT DISTINCT L.COUNTRY_ID,C.COUNTRY_NAME
FROM 
COUNTRIES C INNER JOIN LOCATIONS L
ON C.COUNTRY_ID=L.COUNTRY_ID
INNER JOIN
DEPARTMENTS D
ON D.LOCATION_ID=L.LOCATION_ID)
