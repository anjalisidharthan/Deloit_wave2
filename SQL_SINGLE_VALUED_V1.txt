5)	SELECT  FIRST_NAME,
        MONTHLY_DISCOUNT,
        (MONTHLY_DISCOUNT+MONTHLY_DISCOUNT*.197) AS AFTER_ADDITION,
        ROUND(MONTHLY_DISCOUNT+MONTHLY_DISCOUNT*.197) AS AFTER_ADDITION_WHOLE,
        FLOOR(MONTHLY_DISCOUNT+MONTHLY_DISCOUNT*.197) AS AFTER_ADDITION_FLOOR,
        CEIL(MONTHLY_DISCOUNT+MONTHLY_DISCOUNT*.197) AS AFTER_ADDITION_CEIL
        FROM CUSTOMERS
7)	SELECT FIRST_NAME,BIRTH_DATE,ROUND((MONTHS_BETWEEN(SYSDATE,BIRTH_DATE)/12)) AS AGE

8)	SELECT *
