CREATE TABLE EMPS AS SELECT*FROM EMPLOYEES

ALTER TABLE EMPS ADD(STARS VARCHAR2(50));


DECLARE
V_EMPNO EMPS.EMPLOYEE_ID%TYPE:=176;
V_ASTERIC EMPS.STARS%TYPE:='';
SAL EMPS.SALARY%TYPE;
BEGIN
SELECT NVL(ROUND(SALARY/1000),0) INTO SAL FROM EMPS WHERE EMPLOYEE_ID=V_EMPNO;

FOR I IN 1..SAL LOOP
V_ASTERIC:=V_ASTERIC||'*';
END LOOP;
UPDATE EMPS SET STARS=V_ASTERIC WHERE EMPLOYEE_ID=V_EMPNO;
END;
